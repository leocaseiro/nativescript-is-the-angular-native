### NativeScriptHttpClientModule or NativeScriptHttpModule

```
import {
    NativeScriptHttpModule
} from "nativescript-angular/http";
```

```
import {
    NativeScriptHttpClientModule
} from "nativescript-angular/http-client";

@NgModule({
    ...
    imports: [
        NativeScriptModule,
        NativeScriptHttpClientModule
    ]
})

```
